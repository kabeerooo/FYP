    <script>
        // Simple, clean stock chart animation
        const canvas = document.getElementById('stockAnimation');
        const ctx = canvas.getContext('2d');
        
        // Set canvas size
        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        
        resizeCanvas();
        window.addEventListener('resize', resizeCanvas);
        
        // Stock lines data
        const stockLines = [];
        const numLines = 2;
        
        // Initialize stock lines
        for (let i = 0; i < numLines; i++) {
            const line = {
                points: [],
                color: i === 0 ? 'rgba(29, 161, 242, 0.15)' : 'rgba(29, 161, 242, 0.08)',
                speed: 0.3 + Math.random() * 0.5,
                amplitude: 40 + Math.random() * 30,
                frequency: 0.005 + Math.random() * 0.01,
                offset: Math.random() * 1000
            };
            
            // Generate initial points
            for (let x = 0; x < canvas.width + 200; x += 10) {
                const y = canvas.height / 2 + 
                         Math.sin(x * line.frequency + line.offset) * line.amplitude +
                         Math.sin(x * line.frequency * 0.5 + line.offset) * line.amplitude * 0.3;
                line.points.push({ x, y });
            }
            
            stockLines.push(line);
        }
        
        // Animation function
        function animate() {
            // Clear canvas with slight transparency for trail effect
            ctx.fillStyle = 'rgba(15, 20, 25, 0.02)';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            // Update and draw stock lines
            stockLines.forEach(line => {
                // Update points
                line.points.forEach((point, index) => {
                    point.x -= line.speed;
                    
                    // Regenerate points that have moved off screen
                    if (point.x < -50) {
                        point.x = canvas.width + 50;
                        const y = canvas.height / 2 + 
                                 Math.sin(point.x * line.frequency + line.offset) * line.amplitude +
                                 Math.sin(point.x * line.frequency * 0.5 + line.offset) * line.amplitude * 0.3;
                        point.y = y;
                    }
                });
                
                // Draw line
                ctx.strokeStyle = line.color;
                ctx.lineWidth = 2;
                ctx.beginPath();
                
                line.points.forEach((point, i) => {
                    if (i === 0) {
                        ctx.moveTo(point.x, point.y);
                    } else {
                        ctx.lineTo(point.x, point.y);
                    }
                });
                
                ctx.stroke();
            });
            
            requestAnimationFrame(animate);
        }
        
        // Start animation
        animate();
        
        // Mini charts for assets
        function createMiniChart(canvasId, isPositive) {
            const chartCanvas = document.getElementById(canvasId);
            if (!chartCanvas) return;
            
            const chartCtx = chartCanvas.getContext('2d');
            const width = chartCanvas.width = chartCanvas.offsetWidth * 2;
            const height = chartCanvas.height = chartCanvas.offsetHeight * 2;
            chartCtx.scale(2, 2);
            
            // Generate random data points
            const points = [];
            const numPoints = 20;
            let lastY = height / 4;
            
            for (let i = 0; i < numPoints; i++) {
                const x = (i / (numPoints - 1)) * (width / 2);
                const change = (Math.random() - 0.5) * 10;
                lastY += change;
                
                // Ensure the chart stays within bounds
                lastY = Math.max(5, Math.min(height / 2 - 5, lastY));
                
                points.push({ x, y: lastY });
            }
            
            // Draw the chart
            chartCtx.strokeStyle = isPositive ? 'rgba(23, 191, 99, 0.8)' : 'rgba(224, 36, 94, 0.8)';
            chartCtx.lineWidth = 2;
            chartCtx.beginPath();
            
            points.forEach((point, i) => {
                if (i === 0) {
                    chartCtx.moveTo(point.x, point.y);
                } else {
                    chartCtx.lineTo(point.x, point.y);
                }
            });
            
            chartCtx.stroke();
        }
        
        // Initialize mini charts
        createMiniChart('appleChart', true);
        createMiniChart('teslaChart', false);
        createMiniChart('bitcoinChart', true);
        createMiniChart('nvidiaChart', true);
        createMiniChart('goldChart', true);
    </script>
